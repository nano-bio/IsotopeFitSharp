95  internal void SplineNotAKnot(double[] x, double[] y)
96  {
97  int n = x.Length;
98 
99  if (n < 3)
100  {
101  throw new Exception(); //TODO
102  }
103 
104  // find solution for the quadratic coefficients
105  Matrix<double> M = Matrix<double>.Build.Sparse(n, n);
106  Vector<double> v = Vector<double>.Build.Dense(n);
107 
108  // left boundary
109  M.At(0, 0, x[2] - x[1]);
110  M.At(0, 1, x[0] - x[2]);
111  M.At(0, 2, x[1] - x[0]);
112  v.At(0, 0);
113 
114  // central conditions
115  //a1[i] = x[i + 1] - x[i];
116  //a2[i] = 2 * (x[i + 1] - x[i - 1]);
117  //a3[i] = x[i] - x[i - 1];
118  //b[i] = ;
119 
120  for (int i = 1; i < n - 1; i++)
121  {
122  M.At(i, i - 1, x[i] - x[i - 1]);
123  M.At(i, i, 2 * (x[i + 1] - x[i - 1]));
124  M.At(i, i + 1, x[i + 1] - x[i]);
125  v.At(i, 3 * (y[i + 1] - y[i]) / (x[i + 1] - x[i]) - 3 * (y[i] - y[i - 1]) / (x[i] - x[i - 1]));
126  }
127 
128  // right boundary
129  M.At(n - 1, n - 3, x[n - 1] - x[n - 2]);
130  M.At(n - 1, n - 2, x[n - 3] - x[n - 1]);
131  M.At(n - 1, n - 1, x[n - 2] - x[n - 3]);
132  v.At(n - 1, 0);
133 
134  // solve the system
135  double[] c2 = M.Solve(v).ToArray();
136 
137  // solve for linear and cubic coefficients
138  double[] c1 = new double[n - 1];
139  double[] c3 = new double[n - 1];
140 
141  for (int i = 0; i < n - 1; i++)
142  {
143  c1[i] = (y[i + 1] - y[i]) / (x[i + 1] - x[i]) - (x[i + 1] - x[i]) * (2 * c2[i] + c2[i + 1]) / 3;
144  c3[i] = (c2[i + 1] - c2[i]) / (3 * (x[i + 1] - x[i]));
145  }
146 
147  // constant coefficients are just y values
148  double[] c0 = y.Take(n - 1).ToArray();
149 
150  // save the breaks and coefficient matrix
151  Breaks = x;
152  Coefs = Matrix<double>.Build.DenseOfColumnArrays(new double[][] { c0, c1, c2, c3 }).ToRowArrays();
153  }

166  private void PCHIP(double[] x, double[] y)
167  {
168  /*
169  * Before the interpolation itself, we need to calculate
170  * the derivative values to maintain interval monotonicity and function shape.
171  * Method used is from:
172  * F. N. Fritch, R. E. Carlson, 1980, Monotone Piecewise Cubic Interpolation, SIAM J. Numer. Anal. 17, pp. 238-246
173  * It is also used by GNU Octave.
174  */
175 
176  // number of input and output values
177  int n = x.Length;
178  //int nInterp = xToEval.Length;
179 
180  // variables related to horizontal data
181  double h0, h1, hSum;
182 
183  // variables related to function deltas
184  double delta0, delta1, deltaSave;
185  double deltaMax, deltaMin;
186  double deltaAdj0, deltaAdj1;
187 
188  // weights for calculating the derivative
189  double w0, w1;
190 
191  double[] deriv = new double[n];
192 
193  // array for output values
194  //double[] interpVal = new double[nInterp];
195 
196 
197  h0 = x[1] - x[0];
198  delta0 = (y[1] - y[0]) / h0;
199  deltaSave = delta0;
200 
201  h1 = x[2] - x[1];
202  delta1 = (y[2] - y[1]) / h1;
203 
204  // 3-point formula
205  hSum = h0 + h1;
206 
207  w0 = (h0 + hSum) / hSum;
208  w1 = -h0 / hSum;
209 
210  deriv[0] = w0 * delta0 + w1 * delta1;
211 
212  if (Math.Sign(deriv[0]) * (Math.Sign(delta0)) <= 0)
213  {
214  deriv[0] = 0;
215  }
216  else if (Math.Sign(delta0) * Math.Sign(delta1) < 0)
217  {
218  deltaMax = 3.0 * delta0;
219  if (Math.Abs(deriv[0]) > Math.Abs(deltaMax)) deriv[0] = deltaMax;
220  }
221 
222  // internal points loop
223  for (int i = 1; i < n - 1; i++)
224  {
225  if (i != 1)
226  {
227  h0 = h1;
228  h1 = x[i + 1] - x[i];
229  hSum = h0 + h1;
230  delta0 = delta1;
231  delta1 = (y[i + 1] - y[i]) / h1;
232  }
233 
234  deriv[i] = 0;
235  if (Math.Sign(delta0) * Math.Sign(delta1) < 0)
236  {
237  deltaSave = delta1;
238  continue;
239  }
240 
241  if (Math.Sign(delta0) * Math.Sign(delta1) == 0)
242  {
243  if (delta1 == 0) continue;
244  deltaSave = delta1;
245  continue;
246  }
247 
248  w0 = (hSum + h0) / (3 * hSum);
249  w1 = (hSum + h1) / (3 * hSum);
250 
251  deltaMax = Math.Max(Math.Abs(delta0), Math.Abs(delta1));
252  deltaMin = Math.Min(Math.Abs(delta0), Math.Abs(delta1));
253 
254  deltaAdj0 = delta0 / deltaMax;
255  deltaAdj1 = delta1 / deltaMax;
256 
257  deriv[i] = deltaMin / (w0 * deltaAdj0 + w1 * deltaAdj1);
258  }
259 
260  // Special case - last point
261  w0 = -h1 / hSum;
262  w1 = (h1 + hSum) / hSum;
263  deriv[n - 1] = w0 * delta0 + w1 * delta1;
264 
265  if (Math.Sign(deriv[n - 1]) * Math.Sign(delta1) <= 0)
266  {
267  deriv[n - 1] = 0;
268  }
269  else if (Math.Sign(delta0) * Math.Sign(delta1) < 0)
270  {
271  deltaMax = 3.0 * delta1;
272  if (Math.Abs(deriv[n - 1]) > Math.Abs(deltaMax)) deriv[n - 1] = deltaMax;
273  }
274 
275  // Create a hermite cubic spline interpolation from a set of (x,y) value pairs and their slope (first derivative), sorted ascendingly by x.
276  if (x.Length != y.Length || x.Length != deriv.Length)
277  {
278  throw new ArgumentException(Resources.ArgumentVectorsSameLength);
279  }
280 
281  if (x.Length < 2)
282  {
283  throw new ArgumentException(string.Format(Resources.ArrayTooSmall, 2), "x");
284  }
285 
286  // Constant, liner, quadratic and cubic coefficient, respectively.
287  double[] c0 = new double[x.Length - 1];
288  double[] c1 = new double[x.Length - 1];
289  double[] c2 = new double[x.Length - 1];
290  double[] c3 = new double[x.Length - 1];
291 
292  for (int i = 0; i < c0.Length; i++)
293  {
294  double w = x[i + 1] - x[i];
295  double w2 = w * w;
296  c0[i] = y[i];
297  c1[i] = deriv[i];
298  c2[i] = (3 * (y[i + 1] - y[i]) / w - 2 * deriv[i] - deriv[i + 1]) / w;
299  c3[i] = (2 * (y[i] - y[i + 1]) / w + deriv[i] + deriv[i + 1]) / w2;
300  }
301 
302  Breaks = x;
303  Coefs = Matrix<double>.Build.DenseOfColumnArrays(c0, c1, c2, c3).ToRowArrays();
304  }

341  internal static double PPEval(double[] breaks, double[][] coefs, double x)
342  {
343  // Assigns an index of equal break point or a bitwise compelent index of first greater one to the point x.
344  int breakIndex = Array.BinarySearch(breaks, x);
345 
346  // Selected point equals to one of the break points.
347  if (breakIndex >= 0)
348  {
349  if (breakIndex == breaks.Length - 1)
350  {
351  return MathNet.Numerics.Evaluate.Polynomial((x - breaks[breakIndex - 1]), coefs[breakIndex - 1]);
352  }
353  return coefs[breakIndex][0];
354  }
355  // Selected point is in between of two break points.
356  else
357  {
358  int indexOfGreater = ~breakIndex;
359 
360  if (indexOfGreater == 0) indexOfGreater++;
361  if (indexOfGreater == breaks.Length) indexOfGreater--;
362 
363  if ((0 < indexOfGreater) && (indexOfGreater < breaks.Length))
364  {
365  return MathNet.Numerics.Evaluate.Polynomial((x - breaks[indexOfGreater - 1]), coefs[indexOfGreater - 1]);
366  }
367  else
368  {
369  throw new InterpolationException("Evaluation point is outside of definition range.");
370  }
371  }
