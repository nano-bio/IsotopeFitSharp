138  public void CorrectBaseline(double[] xAxis = null, double[] yAxis = null)
139  {
140  // raw spectral data are required
141  if (SpectralData.RawSignalAxis == null || SpectralData.RawMassAxis == null) throw new WorkspaceNotDefinedException("Raw spectral data not specified.");
142  if (SpectralData.RawSignalAxis.Length != SpectralData.RawMassAxis.Length) throw new WorkspaceNotDefinedException("Supplied spectral data axis have different lengths.");
143 
144  // if there are optional background correction points specified, save them to the workspace and use those
145  if (xAxis != null) BaselineCorrData.XAxis = xAxis;
146  if (yAxis != null) BaselineCorrData.YAxis = yAxis;
147 
148  // safety check, they can still be null in come use cases
149  if (BaselineCorrData.XAxis == null || BaselineCorrData.YAxis == null) throw new WorkspaceNotDefinedException("Baseline correction points not specified.");
150  if (BaselineCorrData.XAxis.Length != BaselineCorrData.YAxis.Length) throw new WorkspaceNotDefinedException("Supplied baseline correction point arrays have different lengths.");
151 
152  int massAxisLength = SpectralData.RawLength;
153 
154  //TODO: Evaluating the bg correction for the whole range might be useless. Specifiyng a mass range would make sense.
155  //TODO: crop the mass axis to the last specified point of the baseline? might break usefulness.
156  PPInterpolation baselineFit = new PPInterpolation(BaselineCorrData.XAxis, BaselineCorrData.YAxis, PPInterpolation.PPType.PCHIP); // in the matlab code it is also hard-coded pchip
157 
158  SpectralData.Baseline = new double[massAxisLength];
159  SpectralData.SignalAxis = new double[massAxisLength];
160 
161  for (int i = 0; i < massAxisLength; i++)
162  {
163  SpectralData.Baseline[i] = baselineFit.Evaluate(SpectralData.RawMassAxis[i]);
164  SpectralData.SignalAxis[i] = SpectralData.RawSignalAxis[i] - SpectralData.Baseline[i];
165  }

180  public void CorrectMassOffset(Interpolation.Type interpType, int order, double[] comList = null, double[] massOffsetList = null)
181  {
182  //TODO: this can be cleaned/optimized, it is an ugly mess right now

201  }
202  }
203  else // this is the second use case for GUI, because only the GUI will supply the optional arguments if needed. IFDFile case just passes over those ifs
204  {
205  if (comList != null) Calibration.COMList = comList;
206  if (massOffsetList != null) Calibration.MassOffsetList = massOffsetList;

211  if (Calibration.COMList.Length != Calibration.MassOffsetList.Length) throw new WorkspaceNotDefinedException("Supplied mass offset correction point arrays have different lengths.");
212

218  double xAxisMax = SpectralData.RawMassAxis.Max();

222  double[] xAxis = new double[xAxisLength];
223

227  }
228

230  Interpolation massOffset;
231 
232  switch (interpType)
233  {
234  case Interpolation.Type.Polynomial:
235  massOffset = new PolyInterpolation(Calibration.COMList.ToArray(), Calibration.MassOffsetList.ToArray(), order);
236  break;
237  case Interpolation.Type.SplineNatural:
238  throw new NotImplementedException();
239  //break;
240  case Interpolation.Type.SplineNotAKnot:

242  break;
243  case Interpolation.Type.PCHIP:
244  massOffset = new PPInterpolation(Calibration.COMList.ToArray(), Calibration.MassOffsetList.ToArray(), PPInterpolation.PPType.PCHIP);

247  throw new Interpolation.InterpolationException("Unknown interpolation type specified.");
248  }

253  //yAxis = massOffset.Evaluate(xAxis.ToArray());

263  double yAxisMax = yAxis.Max(); //TODO: this is not yet robust
264  int yAxisMaxIndex = Array.BinarySearch(yAxis, yAxisMax);
