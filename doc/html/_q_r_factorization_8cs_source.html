200  public static Matrix<double> SparseQRGivens(Matrix<double> M)
201  {
202  Givens gp;
203  double x, y;
204 
205  for (int i = 0; i < M.ColumnCount; i++)
206  {
207  for (int j = M.RowCount - 1; j > i; j--)
208  {
209  if (M.At(j, i) == 0) continue;
210 
211  gp = GivensParams(M.At(i, i), M.At(j, i));
212 
213  for (int k = i; k < M.ColumnCount; k++)
214  {
215  x = M.At(i, k);
216  y = M.At(j, k);
217 
218  M.At(i, k, gp.c * x + gp.s * y);
219  M.At(j, k, -gp.s * x + gp.c * y);
220 
221  //TODO: add zero-coercing
222  }
223  }
224  }
225 
226  return M.SubMatrix(0, M.ColumnCount, 0, M.ColumnCount);
227  }

235  private static Givens GivensParams(double v1, double v2)
236  {
237  Givens gp = new Givens();
238 
239  if (v2 == 0)
240  {
241  gp.c = 1;
242  gp.s = 0;
243  }
244  else if (v1 == 0)
245  {
246  gp.c = 0;
247  gp.s = 1;
248  }
249  else
250  {
251  //TODO: test if putting these operations together is more effective
252  double fg2 = v1 * v1 + v2 * v2;
253  double r = Math.Sqrt(fg2);
254  double rr = 1d / r;
255  gp.c = Math.Abs(v1) * rr;
256  gp.s = v2 * rr;
257 
258  if (v1 < 0) gp.s = -gp.s;
259  }
260 
261  return gp;
262  }
